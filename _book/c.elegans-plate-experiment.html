Scatterplot

# plot dataset in a scatterplot
scatterplot <- CE_LIQ_FLOW_062_Tidydata %>% 
  ggplot(aes(x = log10(compConcentration + .001), y = RawData))+
  geom_point(aes(color = compName, shape = expType),
    size = 1.5, 
    alpha = 0.8,
    position = position_jitter(width = 0.1))+ # jitter to avoid overlap 
  labs(
    title = "Scatterplot CE_LIQ_FLOW_062_Tidydata",
    x = "log10(compConcentration) in nM",
    y = "RawData in counts"
  )

Normalising dataset

# filter on "controlNegative" data
CE_LIQ_FLOW_062_contNeg <- CE_LIQ_FLOW_062_Tidydata %>% filter(expType == "controlNegative")

# calculate mean of "RawData" of the filtered dataset
contNeg_mean <- mean(CE_LIQ_FLOW_062_contNeg$RawData)

# check contNeg_mean
contNeg_mean
## [1] 85.9
# create a normalised column with the normalised values.
CE_LIQ_FLOW_062_Tidydata_norm <- CE_LIQ_FLOW_062_Tidydata %>% mutate(normalised = RawData/mean(contNeg_mean))

# mean RawData
mean(CE_LIQ_FLOW_062_Tidydata_norm$RawData, na.rm=T)
## [1] 68.10423
# check normalised column
CE_LIQ_FLOW_062_Tidydata_norm %>% 
  select(compName, expType, compConcentration, RawData, normalised) %>% 
  reactable(defaultPageSize = 5)
# check the normalisation values are a fraction of 1
CE_LIQ_FLOW_062_contNeg_norm <- CE_LIQ_FLOW_062_Tidydata_norm %>% filter(expType == "controlNegative")

contNegNorm_mean <- mean(CE_LIQ_FLOW_062_contNeg_norm$normalised)

# check if mean value of controlNegative is equal to "1"
contNegNorm_mean
## [1] 1
